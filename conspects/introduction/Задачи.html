Докажите, что следующий код имеет асимптотику $O(n^3)$, посчитав количество операций в зависимости от n. 

long long cnt = 0;
for (int i = 0; i < n; i++) {
    // Выполнится n раз
    for (int j = 0; j < n; j++) {
        // Выполнится n раз
        for (int k = j; k < n; k++) {
            // Выполнится n-j раз
            cnt++;
        }
    }
}
